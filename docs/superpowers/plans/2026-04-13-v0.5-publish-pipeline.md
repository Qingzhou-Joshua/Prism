# Prism v0.5 Publish Pipeline Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Turn the v0.4 dry-run preview into a real publish pipeline — write rule files to platform directories, auto-backup originals, record Revisions, and support full rollback.

**Architecture:** `PublishEngine` in `packages/core` orchestrates file writes and backup creation; `FileRevisionStore` persists per-date JSON records in `~/.prism/revisions/`. The server wires both into new `POST /profiles/:id/publish` and `GET|POST /revisions` routes, and the web frontend gains a Publish flow in `ProfileEditorPage` plus a new `RevisionsPage`.

**Tech Stack:** TypeScript, Vitest (tests), nanoid (revision IDs), Node.js `fs/promises` (file I/O), React + Vite (frontend), Fastify (server).

```ts
import { copyFile, mkdir, writeFile, stat } from 'node:fs/promises'
import { join, basename } from 'node:path'
import { nanoid } from 'nanoid'
import type { PlatformId } from '@prism/shared'
import type { Revision, PublishedFile } from '@prism/shared'
import type { RuleStore } from '../rules/store.js'
import type { ProfileStore } from '../profiles/store.js'
import { getPlatformRulesDir, ruleFileName } from './platform-paths.js'
import { projectRule } from '../rules/project.js'

export class PublishEngine {
  constructor(
    private readonly ruleStore: RuleStore,
    private readonly profileStore: ProfileStore,
    private readonly prismDir: string = (() => {
      const { homedir } = require('node:os')
      return join(homedir(), '.prism')
    })(),
    private readonly getPlatformDir: (platformId: PlatformId) => string = getPlatformRulesDir,
  ) {}

  async publish(profileId: string): Promise<Revision> {
    const profile = await this.profileStore.get(profileId)
    if (!profile) {
      throw new Error(`Profile not found: ${profileId}`)
    }

    // Resolve rules (skip missing)
    const rules = await Promise.all(
      profile.ruleIds.map(id => this.ruleStore.get(id))
    )
    const resolvedRules = rules.filter((r): r is NonNullable<typeof r> => r !== null)

    const revisionId = nanoid()
    const backupsDir = join(this.prismDir, 'backups', revisionId)
    const publishedFiles: PublishedFile[] = []

    for (const platformId of profile.targetPlatforms) {
      const targetDir = this.getPlatformDir(platformId)

      for (const rule of resolvedRules) {
        const fileName = ruleFileName(rule.name)
        const targetPath = join(targetDir, fileName)
        const backupDir = join(backupsDir, platformId)
        const backupPath = join(backupDir, fileName)

        // Check if target already exists
        let isNew = true
        try {
          await stat(targetPath)
          isNew = false
        } catch {
          // file does not exist — isNew stays true
        }

        if (!isNew) {
          // Backup existing file
          await mkdir(backupDir, { recursive: true })
          await copyFile(targetPath, backupPath)
        }

        // Write projected content to target
        await mkdir(targetDir, { recursive: true })
        const content = projectRule(rule, platformId)
        await writeFile(targetPath, content, 'utf8')

        publishedFiles.push({
          platformId,
          filePath: targetPath,
          backupPath,
          isNew,
          ruleId: rule.id,
          ruleName: rule.name,
        })
      }
    }

    return {
      id: revisionId,
      profileId: profile.id,
      profileName: profile.name,
      publishedAt: new Date().toISOString(),
      files: publishedFiles,
    }
  }
}
```

```ts
import { readFile, writeFile, mkdir, readdir, rm, copyFile } from 'node:fs/promises'
import { join } from 'node:path'
import type { Revision } from '@prism/shared'

export class FileRevisionStore {
  private writeQueue: Promise<void> = Promise.resolve()

  constructor(
    private readonly revisionsDir: string,
    private readonly backupsDir: string,
  ) {}

  private dateKey(isoString: string): string {
    return isoString.slice(0, 10) // 'YYYY-MM-DD'
  }

  private dateFilePath(dateKey: string): string {
    return join(this.revisionsDir, `${dateKey}.json`)
  }

  private async loadDateFile(dateKey: string): Promise<Revision[]> {
    try {
      const raw = await readFile(this.dateFilePath(dateKey), 'utf8')
      return JSON.parse(raw) as Revision[]
    } catch {
      return []
    }
  }

  private enqueue(fn: () => Promise<void>): Promise<void> {
    this.writeQueue = this.writeQueue.then(fn)
    return this.writeQueue
  }

  async save(revision: Revision): Promise<void> {
    return this.enqueue(async () => {
      await mkdir(this.revisionsDir, { recursive: true })
      const key = this.dateKey(revision.publishedAt)
      const existing = await this.loadDateFile(key)
      existing.push(revision)
      await writeFile(this.dateFilePath(key), JSON.stringify(existing, null, 2), 'utf8')
    })
  }

  async get(id: string): Promise<Revision | null> {
    const dateKeys = await this.listDateKeys()
    for (const key of dateKeys) {
      const revisions = await this.loadDateFile(key)
      const found = revisions.find(r => r.id === id)
      if (found) return found
    }
    return null
  }

  async listAll(): Promise<Revision[]> {
    const dateKeys = await this.listDateKeys()
    const all: Revision[] = []
    for (const key of dateKeys) {
      const revisions = await this.loadDateFile(key)
      all.push(...revisions)
    }
    return all.sort((a, b) => b.publishedAt.localeCompare(a.publishedAt))
  }

  async rollback(id: string): Promise<void> {
    const revision = await this.get(id)
    if (!revision) {
      throw new Error(`Revision not found: ${id}`)
    }

    for (const file of revision.files) {
      if (file.isNew) {
        // File was created by this publish; delete it on rollback
        try {
          await rm(file.filePath, { force: true })
        } catch {
          // If file is already gone, that's fine
        }
      } else {
        // Restore backup to original location
        await mkdir(join(file.filePath, '..'), { recursive: true })
        await copyFile(file.backupPath, file.filePath)
      }
    }
  }

  private async listDateKeys(): Promise<string[]> {
    try {
      const files = await readdir(this.revisionsDir)
      return files
        .filter(f => f.endsWith('.json'))
        .map(f => f.replace('.json', ''))
        .sort()
    } catch {
      return []
    }
  }
}
```

```ts
    try {
      await revisionStore.rollback(request.params.id)
      return reply.send({ ok: true })
    } catch (err: unknown) {
      const message = err instanceof Error ? err.message : String(err)
      if (message.includes('Revision not found')) {
        return reply.status(404).send({ error: message })
      }
      return reply.status(500).send({ error: message })
```

```ts
import type { Revision } from '@prism/shared'

const API_BASE = 'http://localhost:3001'

async function request<T>(path: string, init?: RequestInit): Promise<T | null> {
  const res = await fetch(`${API_BASE}${path}`, {
    headers: { 'Content-Type': 'application/json' },
    ...init,
  })
  if (res.status === 204) return null
  if (!res.ok) {
    const text = await res.text()
    throw new Error(`HTTP ${res.status}: ${text}`)
  }
  return res.json() as Promise<T>
}

export const revisionsApi = {
  async list(): Promise<Revision[]> {
    const data = await request<{ items: Revision[] }>('/revisions')
    return data?.items ?? []
  },

  async get(id: string): Promise<Revision> {
    const data = await request<Revision>(`/revisions/${id}`)
    if (!data) throw new Error(`Revision ${id} not found`)
    return data
  },

  async rollback(id: string): Promise<void> {
    await request<{ ok: boolean }>(`/revisions/${id}/rollback`, { method: 'POST' })
  },
}

export async function publishProfile(profileId: string): Promise<Revision> {
  const data = await request<{ revision: Revision }>(`/profiles/${profileId}/publish`, {
    method: 'POST',
  })
  if (!data) throw new Error('Publish returned no data')
  return data.revision
}
```
